use std
def scale = comptime { 2^5 }
fn tag(n) case n {
   0..31 -> "small"
   32..63 -> "mid"
   _ -> "big"
}
def lens = [[3,4],[6,8],[5,12],[8,15]].map(fn(p){ int(sqrt(p[0]*p[0] + p[1]*p[1])) })
def sum = lens.reduce(0, fn(a,n){ a + n })
print(f"{scale=} {lens=} {sum=} tag={tag(sum)}")
